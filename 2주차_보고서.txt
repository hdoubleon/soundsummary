[터치 이벤트 핸들러]

TouchStart:
function handleTouchStart(e) {
    if (window.innerWidth > 768) return;
    
    const touch = e.touches[0];
    touchStartY = touch.clientY;
    touchCurrentY = touch.clientY;
    isSidebarDragging = true;
    
    const sidebar = document.querySelector('.sidebar');
    sidebar.classList.add('touch-dragging');
}

TouchMove:
function handleTouchMove(e) {
    if (!isSidebarDragging || window.innerWidth > 768) return;
    
    const touch = e.touches[0];
    touchCurrentY = touch.clientY;
    const deltaY = touchCurrentY - touchStartY;
    
    const sidebar = document.querySelector('.sidebar');
    
    // 아래로 드래그 (닫기)
    if (deltaY > 0 && mobileSidebarOpen) {
        e.preventDefault();
        const translateY = Math.min(deltaY, window.innerHeight);
        sidebar.style.transform = `translateY(${translateY}px)`;
    }
    
    // 위로 드래그 (열기)
    if (deltaY < 0 && !mobileSidebarOpen) {
        e.preventDefault();
        const translateY = Math.max(0, window.innerHeight + deltaY);
        sidebar.style.transform = `translateY(${translateY}px)`;
    }
}

TouchEnd:
function handleTouchEnd(e) {
    if (!isSidebarDragging || window.innerWidth > 768) return;
    
    const deltaY = touchCurrentY - touchStartY;
    const sidebar = document.querySelector('.sidebar');
    
    sidebar.classList.remove('touch-dragging');
    sidebar.style.transform = '';
    
    // 50px 이상 드래그하면 토글
    if (Math.abs(deltaY) > 50) {
        if (deltaY > 0 && mobileSidebarOpen) {
            closeMobileSidebar();
        } else if (deltaY < 0 && !mobileSidebarOpen) {
            openMobileSidebar();
        }
    }
    
    isSidebarDragging = false;
}

6.3 배운 점
--------------------------------------------------------------------------------
[기술적 학습]
1. CSS 변수 시스템
   - :root와 클래스 기반 테마 전환
   - var() 함수 활용
   - 폴백 값 설정

2. 터치 이벤트
   - touchstart, touchmove, touchend
   - passive 이벤트 리스너
   - 제스처 인식 알고리즘

3. 애니메이션 최적화
   - transform vs position 성능 차이
   - will-change 속성
   - 60fps 유지 기법

4. 반응형 디자인
   - 미디어 쿼리 전략
   - 모바일 퍼스트 vs 데스크톱 퍼스트
   - 터치 디바이스 감지

[디자인 원칙]
1. 점진적 향상 (Progressive Enhancement)
   - 기본 기능은 모든 브라우저에서 작동
   - 고급 기능은 지원되는 브라우저에서만

2. 일관성 (Consistency)
   - 전체 UI에서 동일한 색상 체계
   - 통일된 트랜지션 속도

3. 접근성 (Accessibility)
   - 충분한 색상 대비
   - 최소 터치 영역 44x44px
   - 키보드 네비게이션 지원
